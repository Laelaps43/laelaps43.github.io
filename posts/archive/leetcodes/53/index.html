思想

当sum <= 0时，重新将sum赋值为0，因为一个负数加上一个负数，那么他更小，如果他加上一个大于他的数，我们为什么不选择比他的那个数，而需要加上他呢。

Java

public static int maxSubArray(int[] nums) {  
    int max = Integer.MIN_VALUE;  
    int sum = 0;  
    for (int i = 0; i < nums.length; i++) {  
       sum += nums[i];  
       max = Math.max(max, sum);  
       if (sum <= 0)  
           sum = 0;  
    }  
    return max;  
}

Go

func maxSubArray(nums []int) int {
	max := math.MinInt
	sum := 0
	for _, num := range(nums){
		sum += num
		if sum > max { max = sum }
		if sum <= 0 { sum = 0 }
	}
	return max
}
