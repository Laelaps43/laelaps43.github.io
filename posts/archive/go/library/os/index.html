权限

在syscall包下定义了许多关于文件操作的权限的常量。

const (
    O_RDONLY int = syscall.O_RDONLY // 只读模式打开文件
    O_WRONLY int = syscall.O_WRONLY // 只写模式打开文件
    O_RDWR   int = syscall.O_RDWR   // 读写模式打开文件
    O_APPEND int = syscall.O_APPEND // 写操作时将数据附加到文件尾部
    O_CREATE int = syscall.O_CREAT  // 如果不存在将创建一个新文件
    O_EXCL   int = syscall.O_EXCL   // 和O_CREATE配合使用，文件必须不存在
    O_SYNC   int = syscall.O_SYNC   // 打开文件用于同步I/O
    O_TRUNC  int = syscall.O_TRUNC  // 如果可能，打开时清空文件
)

  1. 设置偏移量
// whence == 0 从文件开头开始偏移
// whence == 1 光标所在的位置开始偏移
// whence == 2 从文件末尾开始偏移
func (f *File) Seek(offset int64, whence int) (ret int64, err error)

func TestSeek(t *testing.T) {
    file, err := os.Open("lae.txt")   //打开文件光标默认为文件开头
    if err != nil {
        fmt.Println("err:", err)
    }
    defer file.Close()
    buf := make([]byte, 5)
    file.Seek(3, 0) // 从索引值为3处开始读
    n, err := file.Read(buf)
    if err != nil {
        fmt.Printf("err:%v\n", err)
    }
    fmt.Printf("info:%s\n", string(buf))
    fmt.Printf("n:%d\n", n)
    file.Close()
}
  1. 获取文件的信息
func (f *File) Stat() (FileInfo, error)

func TestStat(t *testing.T) {
    file, err := os.Open("lae.txt")
    if err != nil {
        fmt.Printf("err:%v\n", err)
    }
    defer file.Close()
    fInfo, err := file.Stat()
    if err != nil {
        fmt.Printf("err:%v\n", err)
    }
    fmt.Printf("f是否是一个文件: %v\n", fInfo.IsDir())
    fmt.Printf("f文件的修改时间: %v\n", fInfo.ModTime().String())
    fmt.Printf("f文件的名称: %v\n", fInfo.Name())
    fmt.Printf("f文件的大小: %v\n", fInfo.Size())
    fmt.Printf("f文件的权限: %v\n", fInfo.Mode().String())
}
