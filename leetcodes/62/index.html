前言

在看到这题时，可以很快的想到是使用DP去做，其实我们最先想到的是使用递归去做，但是我感觉有点复杂，因为这个题目的状态转移方程很容易求解出来。

DP

func uniquePaths(m int, n int) int {
    var grid [100][100]int
    grid[0][0] = 1
    for i := 0; i < m; i++{
        for j := 0; j < n; j++{
            if i == 0 && j != 0 {
                // 第一行
                grid[i][j] = grid[i][j - 1]
            }
            if j == 0 && i != 0{
                // 第一列
                grid[i][j] = grid[i - 1][j]
            }
            if i != 0 && j != 0{
                grid[i][j] = grid[i - 1][j] + grid[i][j - 1]
            }
        }
    }
    return grid[m-1][n-1]
}
